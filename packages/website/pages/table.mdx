## Installation
---
### NPM

```bash
npm install @af/styled @af/react-table @af/react-virtual-headless @babel/runtime prop-types --save
```

### Yarn

```bash
yarn add @af/styled @af/react-table @af/react-virtual-headless @babel/runtime prop-types
```

## Table

```jsx
<Table model={model}>
    {i => <div key={i}>{i}</div>}
</Table>
```

### Options

- `model: Model`
  - **Required**
  - return value of `useVirtual()`
- `getRowData: Function(rowIndex: Integer) => Row`
  - **Required**
- `columns: Array`
  - **Required**
- `headless: Boolean`
  - Should render thead or not
- `footer: Boolean`
  - Should render tfoot or not
- `components: Object`
  - See components implementation

- All other props are passed to outermost div

export default ({ children }) => <MdxWrapper>{children}</MdxWrapper>;
