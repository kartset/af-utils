import MdxWrapper from "/components/MdxWrapper";

## Table

```jsx
<Table model={model}>{i => <div key={i}>{i}</div>}</Table>
```

## Installation

```bash
# npm
npm i -S @af-utils/styled @af-utils/react-table @af-utils/react-virtual-headless prop-types

# yarn
yarn add @af-utils/styled @af-utils/react-table @af-utils/react-virtual-headless prop-types
```

### Options

-   `model: Model`
    -   **Required**
    -   return value of `useVirtual()`
-   `getRowData: Function(rowIndex: Integer) => Row`
    -   **Required**
-   `getRowProps: Function(rowData, i) => Object`
    -   Will be spread to row
-   `getKey: Function(i, itemData) => Integer`
    -   Defaults to: `i => i`
    -   if you have unique id to pass as key - it may be more efficient.
-   `columns: Array`
    -   **Required**
-   `headless: Boolean`
    -   Should render thead or not
-   `footer: Boolean`
    -   Should render tfoot or not
-   `components: Object`

    -   See components implementation

-   All other props are passed to outermost div

export default ({ children }) => <MdxWrapper>{children}</MdxWrapper>;
