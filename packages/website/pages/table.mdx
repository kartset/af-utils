## Installation

```bash
# npm
npm i -S @af-utils/styled @af-utils/react-table @af-utils/react-virtual-headless @babel/runtime prop-types

# yarn
yarn add @af-utils/styled @af-utils/react-table @af-utils/react-virtual-headless @babel/runtime prop-types
```

## Table

```jsx
<Table model={model}>{i => <div key={i}>{i}</div>}</Table>
```

### Options

-   `model: Model`
    -   **Required**
    -   return value of `useVirtual()`
-   `getRowData: Function(rowIndex: Integer) => Row`
    -   **Required**
-   `columns: Array`
    -   **Required**
-   `headless: Boolean`
    -   Should render thead or not
-   `footer: Boolean`
    -   Should render tfoot or not
-   `components: Object`

    -   See components implementation

-   All other props are passed to outermost div

export default ({ children }) => <MdxWrapper>{children}</MdxWrapper>;
